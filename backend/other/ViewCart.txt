<?php
session_start();
require_once('db_connection.php');

// Display the contents of the shopping cart
if (isset($_SESSION['cart']) && count($_SESSION['cart']) > 0) {
    echo '<h2>Your Shopping Cart</h2>';
    foreach ($_SESSION['cart'] as $product_id => $quantity) {
        // Fetch product details from the database (replace this with your database logic)
        $product = getProductDetails($product_id);
        
        // Display product information, quantity, etc.
        echo '<p>';
        echo 'Product: ' . $product['name'] . '<br>';
        echo 'Quantity: ' . $quantity . '<br>';
        echo 'Price: $' . $product['price'] * $quantity . '<br>';
        echo '</p>';
    }

    // Display a button to proceed to checkout
    echo '<form action="checkout.php" method="post">';
    echo '<input type="hidden" name="action" value="checkout">';
    echo '<input type="submit" value="Proceed to Checkout">';
    echo '</form>';
} else {
    echo "Your cart is empty.";
}
?>
