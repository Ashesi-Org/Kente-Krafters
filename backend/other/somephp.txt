function getSellers() {
    global $conn;
    $sql = "SELECT DISTINCT seller FROM products";
    $result = $conn->query($sql);

    $sellers = [];

    if ($result->num_rows > 0) {

        }
    }

    return $sellers;
}

    $products = [];
